def bThread(taget):
    SETTHREAD = 800
    print '[Note] Running...\n'
    threadl = []
    queue = Queue.Queue()
    hosts = taget
    for host in hosts:
        queue.put(host)

    threadl = [tThread(queue) for x in xrange(0, int(SETTHREAD))]
    for t in threadl:
        t.start()
    for t in threadl:
        t.join()

class tThread(Thread):

    def __init__(self, queue):
        Thread.__init__(self)
        self.queue = queue

    def run(self):
        while not self.queue.empty():
            host = self.queue.get()
            try:
