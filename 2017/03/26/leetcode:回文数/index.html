title:leetcode:回文数
date:2016-07-09 08:48:52
class Solution {
public:
bool isPalindrome(int x) {
if(x < 0)
return false;
if(x/10 == 0)
return true;

        return isP(toString(x));
    }
    string toString(int x)
    {
        string str("");
        while(x)
        {
            str.push_back(x%10 + '0');
            x/=10;
        }
        return str;
    }
    bool isP(string str)
    {
        int p(0);
        int q(str.size()-1);
        while(p != q && q>=0 && p<=str.size()-1)
        {
            if(str[p] != str[q])
                return false;
            p++;
            q--;
        }
        return true;
    }
};
