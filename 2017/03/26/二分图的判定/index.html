title:二分图的判定
date:2016-05-24 13:49:26

#include <stdio.h>
#include <iostream>
#include <map>
#include <vector>
#include <algorithm>
using std::vector;
using std::map;

std::map<int, std::vector<int> > data;
bool *color;
bool is2f(true);

void init()
{
  // for(int i=0;i<100;i++)
  // {
  //   for(int k=0;k<=100;k++)
  //     {
  //       data[i].push_back(k);
  //     }
  // }
  data[0].push_back(1);
  data[1].push_back(2);
  data[1].push_back(3);
  //data[2].push_back(3);
}


bool *visited;
void dfsColor(int n,bool clr=true)
{

  if(NULL == color)
    color = new bool[data.size()];
  if(NULL == visited)
    visited = new bool[data.size()];
  if(visited[n] == true)
    return;
  else
    visited[n] = true;
  std::cout<<"->"<<n;
  for (int i = 0; i < data[n].size(); ++i)
  {
    if(color[data[n][i]]==clr)
    {
      is2f = false;
      return;
    }
    dfsColor(data[n][i],1-clr);
  }
}

int main(int argc, char const *argv[])
{
  /* code */
  init();
  //dfs(12);
  dfsColor(1);
  std::cout<<std::endl<<is2f;
  return 0;
}

原理：在深度搜索的同时，为每一个节点标记与其父节点不同的颜色，如果遇到相同颜色的节点相邻，则判断为不是二分图，否则是二分图。
