title:leetcode:Reverse Vowels of a String
date:2016-05-15 04:15:41
翻转字符串中的元音

class Solution {
public:
    string reverseVowels(string s) {
        //扫描一遍记录元音位置 下次交换位置
        vector<long> vec;
        size_t len = s.size();
        for(auto x=0;x<len;x++)
        {
            if(s[x]=='a'||s[x]=='e'||s[x]=='i'||s[x]=='o'||s[x]=='u'||s[x]=='A'||s[x]=='E'||s[x]=='I'||s[x]=='O'||s[x]=='U')
                vec.push_back(x);
        }
        len = vec.size();
        for(auto x=0;x<len/2;x++)
        {
            swap(s[vec[x]],s[vec[len-x-1]]);
        }
        return s;
    }
};
