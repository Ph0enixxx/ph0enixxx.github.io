title:leetcode:Missing Number
date:2016-05-29 04:20:03
class Solution {
public:
int missingNumber(vector& nums) {
map numMap;
for(int i=0;i<nums.size();i++)
{
numMap[nums[i]] = true;
}
int i;
auto _max = max_element(nums.begin(),nums.end());
for(i=0;i<=*_max+1;i++)
{
if(numMap[i] == false)
return i;
}
return i;
}
};
