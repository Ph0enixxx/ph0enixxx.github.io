import requests as req
import threading

dirc = []
url = "http://www.ximalaya.com/"
exist = []
#init
def init():
    try:
        print('[+]Init........',end="")
        #read file
        for i in open('./list.txt'):
            if i[-1] == '\n':
                dirc.append(i[:-1])
            else:
                dirc.append(i)
        print('[OK]')
    except e:
        print('[ERROR]')
        print(e)
        exit()
print(dirc)

                    exist.append(tmp.url)
                else:
                    pass
                    #print(Url + "\t\t 404 Not Found")

                #print(e)
                pass
