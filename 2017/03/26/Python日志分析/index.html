d = {}
for i in open('./shadowsocks.log'):
    date = re.search(r'20\d\d-\d+-\d+',i)
    if str(date.group(0)) in d:
        d[str(date.group(0))] += 1
    else:
        d[str(date.group(0))] = 1
print(d)
