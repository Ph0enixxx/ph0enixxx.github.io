title:leetcode: Pascal’s Triangle
date:2016-05-28 01:45:23
杨辉三角，元素等于上一行的两个『肩膀的元素之和』

[

[1],

[1,1],

[1,2,1],

[1,3,3,1],

[1,4,6,4,1]

]

class Solution {
public:
    vector<vector<int>> generate(int numRows) {
        vector<vector<int> > pas;
        if(numRows == 0)
        return pas;
        pas.push_back({1});
        if(numRows == 1)
            return pas;
        pas.push_back({1,1});
        for(int i=2;i<numRows;i++)
        {
            vector<int> t;
            pas.push_back(t);
            pas[i].push_back(1);
            for(int j=1;j<pas[i-1].size();j++)
            {
            pas[i].push_back(pas[i-1][j-1]+pas[i-1][j]);
            }
            pas[i].push_back(1);
        }
        return pas;
    }
};

title:leetcode:Maximum Depth of Binary Tree
date:2016-05-29 05:45:48
/**

 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    int cnt;
    int maxDepth(TreeNode* root) {
        if(NULL == root)
            return 0;
        cnt = 0;
        qianXuMengBi(root);
        return cnt;
    }
    void qianXuMengBi(TreeNode *root,int depth=1)
    {
        if(!root->left && !root->right)
        {
            cnt = max(cnt,depth);
            return;
        }
        if(root->left)
            qianXuMengBi(root->left,depth+1);
        if(root->right)
            qianXuMengBi(root->right,depth+1);

    }
};

title:leetcode:Reverse Vowels of a String
date:2016-05-15 04:15:41
翻转字符串中的元音

class Solution {
public:
    string reverseVowels(string s) {
        //扫描一遍记录元音位置 下次交换位置
        vector<long> vec;
        size_t len = s.size();
        for(auto x=0;x<len;x++)
        {
            if(s[x]=='a'||s[x]=='e'||s[x]=='i'||s[x]=='o'||s[x]=='u'||s[x]=='A'||s[x]=='E'||s[x]=='I'||s[x]=='O'||s[x]=='U')
                vec.push_back(x);
        }
        len = vec.size();
        for(auto x=0;x<len/2;x++)
        {
            swap(s[vec[x]],s[vec[len-x-1]]);
        }
        return s;
    }
};

title:leetcode:Symmetric Tree
date:2016-05-29 05:54:51
/**

 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        if(root == NULL)
            return true;
        return judge(root->left,root->right);
    }
    bool judge(TreeNode* a,TreeNode* b)
    {
        if(!a && !b)
            return true;
        if(!a || !b)
            return false;
        if(a->val != b->val)
            return false;
        bool con1 = judge(a->left,b->right);
        bool con2 = judge(a->right,b->left);
        return con1&&con2;
    }
};

判断二叉树是否为镜像，用递归遍历来判断。

            x/=10;
        }
        return str;
    }

    {
        int p(0);
        int q(str.size()-1);
        while(p != q && q>=0 && p<=str.size()-1)

            if(str[p] != str[q])
                return false;
            p++;
            q--;
        }
        return true;
    }
